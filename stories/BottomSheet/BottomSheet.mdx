## <T k="doc_usage" />

```tsx
import {
  BottomSheet,
  BottomSheetTrigger,
  BottomSheetContent,
  BottomSheetHeader,
  BottomSheetTitle,
  BottomSheetDescription,
  BottomSheetBody,
  BottomSheetFooter,
  BottomSheetClose,
} from "./BottomSheet";

function Example() {
  return (
    <BottomSheet>
      <BottomSheetTrigger>Open</BottomSheetTrigger>
      <BottomSheetContent>
        <BottomSheetHeader>
          <BottomSheetTitle>Title</BottomSheetTitle>
          <BottomSheetDescription>
            Description goes here
          </BottomSheetDescription>
        </BottomSheetHeader>
        <BottomSheetBody>Main content goes here...</BottomSheetBody>
        <BottomSheetFooter>
          <BottomSheetClose>Close</BottomSheetClose>
        </BottomSheetFooter>
      </BottomSheetContent>
    </BottomSheet>
  );
}
```
