  <DropdownTrigger>
    <Button label="Open Menu" />
  </DropdownTrigger>
  <DropdownMenu>
    <DropdownItem onClick={handleProfile}>Profile</DropdownItem>
    <DropdownItem onClick={handleSettings}>Settings</DropdownItem>
  </DropdownMenu>
</Dropdown>;
